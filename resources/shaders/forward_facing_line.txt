#version 450 core

//uniform hits every vertex shader

layout (location = 0) in vec4 offset;
layout (location = 1) uniform mat4 rot;
layout (location = 2) uniform vec2 scaling;

void main (void){
    const vec4 verts[2] = vec4[2](
        vec4(1, 0, 0.1, 1),
        vec4(0, 0, 0.1, 1)
    );

    //Rotate and add offset
    gl_Position = rot * verts[gl_VertexID] + offset;
}
